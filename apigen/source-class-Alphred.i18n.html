 5: /*
 6:  * For internationalization:
 7:  *
 8:  * gettext() isn't compiled into OS X's PHP installation, so I've created my own hacky version.
 9:  *
10:  * (1) Create a directory called "i18n" in your workflow folder.
11:  * (2) For each language you want to use, create a file called ln.json (lowercase) where
12:  * "ln" is the two-letter language code: i.e. en = English, de = German, etc...
13:  * (3) In that json, make the original string as the key and the value as the translated test.
14:  *     i.e., a file called i18n/fr.json would contain:
15:  *     {
16:  *       "Hello": "Bonjour",
17:  *       "Do you speak French?": "Parlez-vous Fran√ßais?",
18:  *       "I am a grapefruit": "Je suis un pamplemousse",
19:  *     }
20:  *
21:  * (4) Make sure you escape the string if necessary.
22:  *
23:  * Tip: if you'd rather have the json created for you, then just do something like...
24:  *         file_put_contents( 'i18n/fr.json', json_encode( [
25:  *             'string'  => 'translated',
26:  *             'string2' => 'second translation'
27:  *         ], JSON_PRETTY_PRINT );
28:  *
29:  *  Then just execute that PHP, and the json file will be created for you.
30:  *
31: **/
32: 
33: class i18n {
34: 
35:     public function __construct() {
36:         if ( ! file_exists( 'i18n' ) || ! is_dir( 'i18n' ) ) { return false; }
37:         if ( ! defined( 'ALPHRED_TESTING' ) || ( true !== ALPHRED_TESTING ) ) {
38:             $locale = exec( 'defaults read .GlobalPreferences AppleLanguages | tr -d [:space:] | cut -c2-3' );
39:         } else {
40:             $locale = 'fr';
41:         }
42:         if ( file_exists( "i18n/{$locale}.json" ) ) {
43:             $this->locale = $locale;
44:         } else {
45:             return false;
46:         }
47: 
48:         try {
49:             $this->dictionary = json_decode( file_get_contents( "i18n/{$locale}.json" ), true );
50:         } catch ( Exception $e ) {
51:             file_put_contents( 'php://stderr', "Error: locale '{$locale}.json' file is not valid json." );
52:             return false;
53:         }
54: 
55:         return $this;
56:     }
57: 
58:     public function translate( $string ) {
59:         if ( ! isset( $this->locale ) ) { return $string; }
60:         if ( isset( $this->dictionary[ $string ] ) ) {
61:             return $this->dictionary[ $string ];
62:         }
63:         return $string;
64:     }
65: }
