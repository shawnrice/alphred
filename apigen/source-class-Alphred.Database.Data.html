 59:     /**
 60:      * Creates the data table
 61:      *
 62:      * Note: this library is meant to store JSON strings as data. So,
 63:      * the data table is simply a key-val storage that the index can
 64:      * then read. You can use this independent of the Index class, but
 65:      * if you are using the Index class, then make sure you feed this
 66:      * JSON.
 67:      *
 68:      * @return boolean      success of sqlite query
 69:      */
 70:     private function create_table() {
 71:         $schema = 'CREATE TABLE IF NOT EXISTS
 72:                     data(
 73:                         id TEXT PRIMARY KEY,
 74:                         time INTEGER NOT NULL,
 75:                         data TEXT NOT NULL
 76:                     );';
 77:         return $this->db->exec( $schema );
 78:     }
 79: 
 80:     /**
 81:      * Creates the internal meta table
 82:      *
 83:      * @return boolean      success of sqlite query
 84:      */
 85:     private function create_meta() {
 86:         $schema = 'CREATE TABLE IF NOT EXISTS
 87:                     meta(
 88:                         key TEXT PRIMARY KEY,
 89:                         value TEXT NOT NULL
 90:                     );';
 91:         return $this->db->exec( $schema );
 92:     }

 94:     /**
 95:      * Adds a record to the data table
 96:      *
 97:      * Note: if you do not provide a unique identifier for the record, then
 98:      * one will be generated as a hash of the record, but it will be harder
 99:      * to curate the data table without one.
100:      *
101:      * @param string    $record     ideally, a string of JSON
102:      * @param string    $unique     optional: the unique identifier for the record
103:      * @param boolean   $overwrite  optional: whether or not to overwrite a record
104:      */
105:     public function add( $record, $unique = false, $overwrite = false ) {
106:         if ( false === $unique ) {
107:             $unique = md5( $record );
108:         }
109:         if ( $overwrite ) {
110:             // We're going to overwrite, but, in case we don't have a good identifier, then
111:             // we'll just delete the field first rather than using an INSERT OR REPLACE query
112:             $this->db->exec(
113:                 'DELETE FROM data WHERE id=\'' . $this->db->escapeString( $unique ) . '\';', true
114:             );
115: 
116:         } else if ( $result = $this->get( $unique ) ) {
117:             // Since we're not
118:             return $result;
119:         }
120:         $this->set_meta( 'update', time() );
121:         $query = 'INSERT INTO data (id, time, data) VALUES (' .
122:                  '\'' . $this->db->escapeString( $unique ) . '\', ' .
123:                  '\'' . time() . '\', ' .
124:                  '\'' . $this->db->escapeString( json_encode( $record ) ) . '\');';
125: 
126:         // Returns true on success false on error. We can build in error-checking later
127:         return $this->db->exec( $query );
128:     }

130:     /**
131:      * Get a specific record from the data table
132:      *
133:      * @param  string   $unique     the unique identifier of the record
134:      * @return string               a string (probably JSON) that has the data
135:      */
136:     public function get( $unique ) {
137:         $unique = $this->db->escapeString( $unique );
138:         $result = $this->db->querySingle(
139:             "SELECT data from data WHERE id='{$unique}';", true
140:         );
141:         if ( 0 === count( $result ) ) {
142:             return false;
143:         }
144:         return $result;
145:     }

147:     /**
148:      * Deletes a record from the data db
149:      *
150:      * @param  string   $unique     the unique identifier for a data record
151:      * @return boolean              returns false on failure
152:      */
153:     public function delete( $unique ) {
154:         if ( empty( $unique ) ) {
155:             return false;
156:         }
157:         $unique = $this->db->escapeString( $unique );
158:         $this->db->exec( "DELETE FROM data WHERE id='{$unique}';" );
159: 
160:         // If a record was actually deleted, then update the last updated
161:         // time on the meta table
162:         if ( $this->db->changes() ) {
163:             $this->set_meta( 'update', time() );
164:         }
165:     }

167:     /**
168:      * Set the value of a key on the meta table
169:      *
170:      * @param string $key name of the key
171:      * @param mixed  $value  the value of the key
172:      */
173:     public function set_meta( $key, $value ) {
174:         $key   = $this->db->escapeString( $key );

176:         return $this->db->exec(
177:             "INSERT OR REPLACE INTO meta(key, value) VALUES ('{$key}', '{$value}');"
