144:     private function log_update( $data_ids, $type ) {
145:         if ( ! in_array( $type, [ 'add', 'delete' ] ) ) {
146:             // Throw an exception really.

155:         if ( is_array( $data_ids ) ) {
156:             $data_ids = implode( ',', $data_ids );
157:         }

160:         $data_ids = $this->index->escapeString( $data_ids );

176:     public function read_updates( $since = 0, $type = false ) {

178:         // Wrangle "since" into a time

189:             $type = false;

453:     /**

459:      * @param  string   $type   SQLite data type for the stored data (TEXT, INTEGER, NUMERIC, etc...)

462:         // Upcase the type
463:         $type = strtoupper( $type );
464: 
465:         // Create the main table
466:         $this->index->exec(
467:             "CREATE TABLE IF NOT EXISTS {$table} (
468:                 data_id     TEXT        NOT NULL,

470:                 FOREIGN KEY (data_id) references data(data_id) ON DELETE CASCADE

514:     /**

516:      *

520:      * @param [type] $data_id [description]
521:      * @param [type] $data    [description]
522:      */
523:     private function add_record( $data_id, $data ) {
524:         if ( ! is_array( $data ) ) {
525:             $data = json_decode( $data, true );
526:             // Add an exception if fails...
527:         }
528: 
529:         // Make sure that the data tables exist for the object
530:         $this->create_index_tables( $data );
531: 
532:         // Add the data to the index
533:         // Basically, what we're going to do here is to walk through the
534:         // data array and consider each key a table, and we're going to push
535:         // value to that table. For arrays, we'll push each value to the table;
536:         // however, we ignore multi-dimensional arrays.
537:         foreach ( $data as $table => $value ) :
538:             if ( ! is_array( $value ) ) {
539:                 $this->insert( $table, $data_id, $value );
540:             } else {
541:                 foreach ( $value as $v ) :
542:                     // Exit if value is also an array
543:                     if ( is_array( $v ) ) {
544:                         continue; }
545:                     $this->insert( $table, $data_id, $v );
546:                 endforeach;
547:             }
548:         endforeach;
549: 
550:         // Log the update
551:         $this->log_update( $data_id, 'add' );
