 17:         $data = [];
 18:         $legend = [
 19:         'millenium' => [ 'multiple' => 'millenia',  'value' => 31536000000 ],
 20:         'century'   => [ 'multiple' => 'centuries', 'value' => 3153600000  ],
 21:         'decade'    => [ 'multiple' => 'decades',   'value' => 315360000   ],
 22:         'year'      => [ 'multiple' => 'years',     'value' => 31536000    ],
 23:         'month'     => [ 'multiple' => 'months',    'value' => 2592000     ],
 24:         'week'      => [ 'multiple' => 'weeks',     'value' => 604800      ],
 25:         'day'       => [ 'multiple' => 'days',      'value' => 86400       ],
 26:         'hour'      => [ 'multiple' => 'hours',     'value' => 3600        ],
 27:         'minute'    => [ 'multiple' => 'minutes',   'value' => 60          ],
 28:         'second'    => [ 'multiple' => 'seconds',   'value' => 1           ]
 29:         ];
 30: 
 31:         // Start with the greatest values and whittle down until we're left with seconds
 32:         foreach ( $legend as $singular => $values ) :
 33:             // If the seconds is greater than the unit, then do some math
 34:             if ( $seconds >= $values['value'] ) {
 35:                 // How many units are in those seconds?
 36:                 $value = floor( $seconds / $values['value'] );
 37:                 if ( $words ) {
 38:                     // We want words, not numbers, so convert to words
 39:                     $value = Date::convert_number_to_words( $value );
 40:                 }
 41:                 // Did we get single or multiple?
 42:                 if ( $seconds / $values['value'] >= 2 ) {
 43:                     // Use plural units
 44:                     $data[ $values['multiple'] ] = $value;
 45:                 } else {
 46:                     // Use singlur units
 47:                     $data[ $singular ] = $value;
 48:                 }
 49:                 // Remove what we just converted and continue
 50:                 $seconds = $seconds % $values['value'];
 51:             }
 52:         endforeach;

 54:         // If we want this as an array, then return that
 55:         if ( 'array' == $type ) {

 94:             15                  => 'fifteen',
 95:             16                  => 'sixteen',
 96:             17                  => 'seventeen',
 97:             18                  => 'eighteen',

 99:             20                  => 'twenty',
100:             30                  => 'thirty',

102:             50                  => 'fifty',
103:             60                  => 'sixty',
104:             70                  => 'seventy',
105:             80                  => 'eighty',
106:             90                  => 'ninety',
107:             100                 => 'hundred',
108:             1000                => 'thousand',
109:             1000000             => 'million',
110:             1000000000          => 'billion',
111:             1000000000000       => 'trillion',
112:             1000000000000000    => 'quadrillion',
113:             1000000000000000000 => 'quintillion'
114:         ];
115: 
116:         if ( ! is_numeric( $number ) ) {
117:             // You didn't feed this a number
118:             return false;
119:         }
120: 
121:         if ( ( $number >= 0 && (int) $number < 0 ) || (int) $number < 0 - PHP_INT_MAX ) {
122:             // overflow
123:             trigger_error(
124:                 'convert_number_to_words only accepts numbers between -' . PHP_INT_MAX . ' and ' . PHP_INT_MAX,
125:                 E_USER_WARNING
126:             );
127:             return false;
128:         }
129: 
130:         if ( $number < 0 ) {
131:             // The number is negative, so re-run the function with the positive value but prepend the negative sign
132:             return $negative . Date::convert_number_to_words( abs( $number ) );
133:         }
134: 
135:         $string = $fraction = null;
136: 
137:         if ( false !== strpos( $number, '.' ) ) {
138:             list( $number, $fraction ) = explode( '.', $number );
139:         }
140: 
141:         // We're going to run through what we have now
142:         switch ( true ) {
143:             case $number < 21:
144:                 $string = $dictionary[ $number ];
145:                 break;
146:             case $number < 100:
147:                 $tens   = ( (int) ( $number / 10 ) ) * 10;
148:                 $units  = $number % 10;
149:                 $string = $dictionary[ $tens ];
150:                 if ( $units ) {

152:                 }
153:                 break;
154:             case $number < 1000:
155:                 $hundreds  = $number / 100;
156:                 $remainder = $number % 100;
157:                 $string = $dictionary[ $hundreds ] . ' ' . $dictionary[100];

159:                     // We have some leftover number, so let's run the function again on what's left

161:                 }
162:                 break;
163:             default:
164:                 $baseUnit = pow( 1000, floor( log( $number, 1000 ) ) );
165:                 $numBaseUnits = (int) ( $number / $baseUnit );

167:                 $string = Date::convert_number_to_words( $numBaseUnits ) . ' ' . $dictionary[ $baseUnit ];
168:                 if ( $remainder ) {

172:                 }
173:                 break;
174:         }
175:
